Restify8（Nodejs）+ Angular7 + PostgreSQL10 搭建个人博客

项目依赖的js库安装

backend:

    npm install *:
        restify:        version 8
        sequelize:      version 5 //数据库ORM
        pg:             version 7 
        pg-hstore:      version 2
        jsonwebtoken:   version 8 //jwt
        uuid:           version 3

frontend:

    npm install *:
        angular-cli:    version 7
        ckeditor:       version 5
        moment:         version 2
        bootstrap:      version 4
        ngx-bootstrap:  version 7

SQL:

PostgreSQL 10.x

创建服务器

var server = restify.createServer();

添加框架自带的一些中间件

server.pre(restify.plugins.pre.userAgentConnection());
server.use(restify.plugins.bodyParser());
server.use(restify.plugins.acceptParser(server.acceptable));
server.use(restify.plugins.authorizationParser());
server.use(restify.plugins.dateParser());
server.use(restify.plugins.queryParser());
server.use(restify.plugins.urlEncodedBodyParser());

添加一些的header

server.use(function(req, res, next){
    res.header("Access-Control-Allow-Origin", 
        "http://127.0.0.1:4200");
    res.header("Access-Control-Expose-Headers", 
        "X-XSRF-TOKEN");
    res.header("Access-Control-Allow-Methods", 
        "POST, GET, PUT, DELETE, OPTIONS");
    res.header("Access-Control-Max-Age", "3600");
    res.header("Access-Control-Allow-Headers", 
        "X-XSRF-TOKEN, Content-Type");
    res.header("Access-Control-Allow-Credentials", "false");
    res.header("Content-Type", 
        "application/json;multipart/form-data;");//upload
    return next();
});
